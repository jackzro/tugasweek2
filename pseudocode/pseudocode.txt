/* Laundry */

START 
    SET pakaian to 0
        FOR pakaian 0 to 20 
            INCREMENT pakaian by 1
                IF (pakaian EQUAL 20)
                    THEN
                    PRINT pakaian sudah dimasukkan semua 
                ELSE
                    PRINT pakaian belum dimasukkan semua 
                END IF
        END FOR 
END
